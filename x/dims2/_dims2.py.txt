    pass

    pass

    pass


class IntShape(BaseShape[int], lang.Final):
    pass


class BaseStrides(BaseDims[SymIntT], lang.Abstract, ta.Generic[SymIntT]):
    pass


class Strides(BaseStrides[SymInt], lang.Final):
